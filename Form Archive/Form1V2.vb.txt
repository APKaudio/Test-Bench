    Public Sub DMM_Read_1()
        'read the value
        DMM_Value_1.Text = DMM1.ReadNumber()

        'if it's Vac calculate dBm
        If DMM_function_1.SelectedIndex = 2 Then
            DMM_dbm_1.Text = 10 * Math.Log10((CDbl(DMM_Value_1.Text) ^ 2) / CDbl(dBm_ref.Text) / (0.001))
        Else
            DMM_dbm_1.Text = ""
        End If
    End Sub
